def branches = []
def gitUrl = 'https://github.com/your/repo.git'  // Replace with your Git repository URL
def gitCredentialsId = 'your-git-credentials-id'  // Replace with your Git credentials ID

def cmd = """
git ls-remote --heads $gitUrl
"""

def gitBranches = bat(script: cmd, returnStatus: true, credentialsId: gitCredentialsId).trim()

if (gitBranches) {
    branches = gitBranches.readLines().collect { it.split()[-1] }
}

return branches.join('\n')
